<template>
  <div class="h-screen flex flex-col overflow-hidden">
    <!-- Header + Tabs -->
    <transition name="slide-down">
      <div v-if="showNav" class="topbar bg-white shadow-md">
        <!-- AppBar -->
        <div class="flex items-center justify-between px-4 py-2 border-b">
          <span class="font-bold">Clinic BS Khang</span>
          <button @click="logout" class="text-red-500">Logout</button>
        </div>

        <!-- Line tím -->
        <div class="bg-purple-500 h-1"></div>

        <!-- Tabs -->
        <div class="flex border-b text-center">
          <button
            class="flex-1 py-2"
            @click="tab = 'tieptan'"
            :class="{ 'font-bold text-purple-600': tab === 'tieptan' }"
          >
            Tiếp tân
          </button>
          <button
            class="flex-1 py-2"
            @click="tab = 'dscho'"
            :class="{ 'font-bold text-purple-600': tab === 'dscho' }"
          >
            DS Chờ
          </button>
          <button
            class="flex-1 py-2"
            @click="tab = 'thongke'"
            :class="{ 'font-bold text-purple-600': tab === 'thongke' }"
          >
            Thống kê
          </button>
        </div>
      </div>
    </transition>

    <!-- Nội dung chính -->
    <div 
      class="flex-1 overflow-y-auto bg-gray-50"
      @scroll.passive="onScroll"
      ref="scrollArea"
    >
      <div v-if="tab === 'tieptan'" class="p-4 space-y-4">
        <p v-for="i in 30" :key="i">📋 Item Tiếp tân số {{ i }}</p>
      </div>

      <div v-if="tab === 'dscho'" class="p-4 space-y-4">
        <p v-for="i in 30" :key="i">📋 Item DS Chờ số {{ i }}</p>
      </div>

      <div v-if="tab === 'thongke'" class="p-4 space-y-4">
        <p v-for="i in 30" :key="i">📊 Item Thống kê số {{ i }}</p>
      </div>
    </div>
  </div>
</template>

<script setup>
import { ref } from 'vue'

const tab = ref('tieptan')
const showNav = ref(true)
let lastScroll = 0
const scrollArea = ref(null)

const onScroll = () => {
  const current = scrollArea.value.scrollTop
  if (current > lastScroll && current > 50) {
    // Vuốt lên => ẩn navigation
    showNav.value = false
  } else if (current < lastScroll) {
    // Vuốt xuống => hiện lại
    showNav.value = true
  }
  lastScroll = current
}

const logout = () => {
  alert('Đăng xuất!')
}
</script>

<style>
/* Transition mượt cho AppBar + Tabs */
.slide-down-enter-active,
.slide-down-leave-active {
  transition: transform 0.3s ease;
}
.slide-down-enter-from,
.slide-down-leave-to {
  transform: translateY(-100%);
}
</style>
